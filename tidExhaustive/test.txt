# --- 高精度计时函数 ---
FUNC PreciseWait
    $f_start = TIME()
    FOR
        $f_now = TIME()
        $f_diff = $f_now - $f_start
        IF $f_diff >= $delay_ms
            BREAK
        ENDIF
    NEXT
    $error = $f_diff - $delay_ms
ENDFUNC

FUNC 匹配
    # ========自定义ID匹配=======
    $is_match = 0
    IF $curr1 == $target1
        IF $curr2 == $target2
            IF $curr3 == $target3
                IF $curr4 == $target4
                    IF $curr5 == $target5
                        $is_match = 1
                    ENDIF
                ENDIF
            ENDIF
        ENDIF
    ENDIF
    
    IF $65535开关 == 1
        IF $curr1 == 6
            IF $curr2 == 5
                IF $curr3 == 5
                    IF $curr4 == 3
                        IF $curr5 == 5
                            $is_match = 1
                        ENDIF
                    ENDIF
                ENDIF
            ENDIF
        ENDIF
    ENDIF
    
    # ========豹子号判断=======
    $same_condition_match = 0
    IF $is_match == 0
        IF $same_id_switch == 1
            IF $curr1 == $curr2
                IF $curr2 == $curr3
                    IF $curr3 == $curr4
                        IF $curr4 == $curr5
                            $same_condition_match = 1
                        ENDIF
                    ENDIF
                ENDIF
            ENDIF
        ENDIF
    ENDIF
    
    # ========连号判断（升序 / 降序）=======
    $continue_condition_match = 0
    IF $continue_id_switch == 1
        IF $is_match == 0
            IF $same_condition_match == 0
                
                # 升序
                $t1 = $curr1
                $t1 += 1
                $t2 = $curr2
                $t2 += 1
                $t3 = $curr3
                $t3 += 1
                $t4 = $curr4
                $t4 += 1
                IF $curr2 == $t1
                    IF $curr3 == $t2
                        IF $curr4 == $t3
                            IF $curr5 == $t4
                                $continue_condition_match = 1
                            ENDIF
                        ENDIF
                    ENDIF
                ENDIF
                
                # 降序
                IF $continue_condition_match == 0
                    $t1 = $curr1
                    $t1 -= 1
                    $t2 = $curr2
                    $t2 -= 1
                    $t3 = $curr3
                    $t3 -= 1
                    $t4 = $curr4
                    $t4 -= 1
                    IF $curr2 == $t1
                        IF $curr3 == $t2
                            IF $curr4 == $t3
                                IF $curr5 == $t4
                                    $continue_condition_match = 1
                                ENDIF
                            ENDIF
                        ENDIF
                    ENDIF
                ENDIF
                
            ENDIF
        ENDIF
    ENDIF
ENDFUNC

FUNC NextSeed
    # 拆 seed
    $hi = $seed >> 16
    $lo = $seed & 65535
    
    # 乘法拆分
    $t1 = $lo * 20077# lo * A_lo
    $t2 = $lo * 16838# lo * A_hi
    $t3 = $hi * 20077# hi * A_lo
    
    # 合并中间项
    $mid = $t2 + $t3
    
    # 组合结果
    $new_lo = $t1 & 65535
    $new_hi = ($mid + ($t1 >> 16)) & 65535
    
    # 加上常数 C
    $new_lo = $new_lo + 24691
    
    IF $new_lo >= 65536
        $new_lo = $new_lo - 65536
        $new_hi = $new_hi + 1
    ENDIF
    
    $new_hi = $new_hi & 65535
    
    # 拼回 32 位
    $seed = ($new_hi << 16) + $new_lo
ENDFUNC
